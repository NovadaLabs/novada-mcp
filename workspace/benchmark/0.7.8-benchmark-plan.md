// Part 1 entry point
FOR EACH test IN [V1..V10]:

### Execution SOP (Plywood Protocol)

```pseudocode
CONST COMPETITORS = ["novada", "firecrawl", "tavily", "oxylabs"]
CONST TASKS = ["T1", "T2", "T3", "T4", "T5", "T6", "T7", "T8"]
CONST ROUNDS = 50
CONST TIMEOUT_MS = 60000
CONST INTER_ROUND_DELAY_MS = 500

// N/A matrix — true = skip all rounds
CONST NA_MATRIX = {
  firecrawl: { T3: true, T4: true, T6: true },
  tavily:    { T5: true, T6: true },
  oxylabs:   { T5: true },
}

FUNCTION run_part2():
  all_results = {}

  FOR EACH competitor IN COMPETITORS:
    PRINT("═" * 60)
    PRINT("  Competitor: {competitor}")
    PRINT("═" * 60)

    rows = []
    credit_exhausted = false

    FOR EACH task_id IN TASKS:
      target = TARGETS[task_id]

      // Skip N/A combinations
      IF NA_MATRIX[competitor]?.[task_id]:
        FOR r IN 0..ROUNDS:
          rows.PUSH(na_record(competitor, task_id, r))
        CONTINUE

      // Stop if credits ran out
      IF credit_exhausted:
        FOR r IN 0..ROUNDS:
          rows.PUSH(credit_exhausted_record(competitor, task_id, r))
        CONTINUE

      PRINT("  {task_id} — {target.label}")

      // Warmup (round 0, discarded from stats)
      warmup = measure(task_id, competitor, 0, () => call_competitor(competitor, task_id, target))
      IF warmup.status == "credit_exhausted":
        credit_exhausted = true

      // 50 measured rounds
      FOR r IN 1..ROUNDS:
        IF credit_exhausted:
          rows.PUSH(credit_exhausted_record(competitor, task_id, r))
          CONTINUE

        IF r > 1: SLEEP(INTER_ROUND_DELAY_MS)

        record = measure(task_id, competitor, r, () => call_competitor(competitor, task_id, target))
        rows.PUSH(record)

        // Novada-only: track 0.7.8 features
        IF competitor == "novada" AND record.success:
          record.v078 = {
            has_source_field:     BOOL(record.text.match(/source:\s*(live|wayback|cache)/)),
            has_chainable_output: record.text.contains("## Chainable Output") OR record.text.contains('"agent_instruction"'),
            has_remember_hint:    record.text.contains("remember:"),
            source_value:         (record.text.match(/source:\s*(live|wayback|cache)/) || [])[1] || "unknown",
            fast_path_used:       task_id IN ["T1", "T7"] AND record.latency_ms < 1500,
          }

        IF record.status == "credit_exhausted":
          credit_exhausted = true
          PRINT("  Credit exhausted at round {r}")

    // Save per-competitor results
    WRITE("results-{competitor}.json", rows.filter(r => r.round > 0))
    all_results[competitor] = rows

  // Aggregate
  summary = aggregate(all_results)
  WRITE("summary.json", summary)

  // Generate report
  report_html = generate_report(summary, all_results)
  WRITE("report-078.html", report_html)

  evidence_md = generate_evidence(all_results)
  WRITE("benchmark-evidence-078.md", evidence_md)

  RETURN summary
```

### Measurement Wrapper

```pseudocode
FUNCTION measure(task_id, competitor, round, api_call_fn):
  t0 = NOW()
  TRY:
    { text, raw } = RACE(api_call_fn(), TIMEOUT(TIMEOUT_MS))
    latency_ms = NOW() - t0
    content_quality = score_content(task_id, text)
    af = score_agent_friendliness(competitor, raw, text, is_error=false)

    RETURN {
      competitor, category: task_id, round, latency_ms,
      success: true, status: "ok",
      content_length_chars: text.length,
      content_quality, agent_friendliness: af,
      target_content_found: content_quality >= 3,
      text: text,  // kept for 0.7.8 feature scanning
      notes: "",
    }

  CATCH err:
    latency_ms = NOW() - t0
    is_credit = err.match(/402|insufficient credits|quota/)
    is_timeout = err.message == "TIMEOUT"
    status = is_credit ? "credit_exhausted" : is_timeout ? "timeout" : "error"

    RETURN {
      competitor, category: task_id, round,
      latency_ms: is_timeout ? TIMEOUT_MS : latency_ms,
      success: false, status,
      content_length_chars: 0, content_quality: 1,
      agent_friendliness: { score: 0 },
      target_content_found: false,
      text: "",
      notes: err.message[0:200],
    }
```

### Aggregation

```pseudocode
FUNCTION aggregate(all_results):
  summary = {}

  FOR EACH competitor IN COMPETITORS:
    summary[competitor] = {}
    rows = all_results[competitor]

    FOR EACH task_id IN TASKS:
      cat_rows = rows.filter(r => r.category == task_id AND r.round > 0)
      success_rows = cat_rows.filter(r => r.status == "ok")
      na_rows = cat_rows.filter(r => r.status == "na")
      latencies = success_rows.map(r => r.latency_ms)
      qualities = success_rows.map(r => r.content_quality)
      af_scores = success_rows.map(r => r.agent_friendliness.score)

      summary[competitor][task_id] = {
        status: na_rows.length == cat_rows.length ? "na"
              : success_rows.length > 0 ? "ok" : "failed",
        success_count: success_rows.length,
        total_rounds: cat_rows.filter(r => r.status != "na").length,
        success_rate: success_rows.length / MAX(1, cat_rows.filter(r => r.status != "na").length),
        latency_median_ms: MEDIAN(latencies),
        latency_p95_ms: P95(latencies),
        quality_median: MEDIAN(qualities),
        af_score: MEDIAN(af_scores),
      }

    // Novada-only: 0.7.8 feature adoption rates
    IF competitor == "novada":
      novada_success = rows.filter(r => r.success AND r.v078)
      summary.novada._v078 = {
        source_field_rate:     novada_success.filter(r => r.v078.has_source_field).length / MAX(1, novada_success.length),
        chainable_output_rate: novada_success.filter(r => r.v078.has_chainable_output).length / MAX(1, novada_success.length),
        remember_hint_rate:    novada_success.filter(r => r.v078.has_remember_hint).length / MAX(1, novada_success.length),
        fast_path_rate_t1:     rows.filter(r => r.category == "T1" AND r.v078?.fast_path_used).length / MAX(1, rows.filter(r => r.category == "T1" AND r.success).length),
        fast_path_rate_t7:     rows.filter(r => r.category == "T7" AND r.v078?.fast_path_used).length / MAX(1, rows.filter(r => r.category == "T7" AND r.success).length),
      }

  RETURN summary
```

### Winner Determination

```pseudocode
FUNCTION determine_winners(summary):
  winners = {}

  FOR EACH task_id IN TASKS:
    best = { quality: -1, latency: Infinity, competitor: null }

    FOR EACH c IN COMPETITORS:
      s = summary[c][task_id]
      IF s.status == "na": CONTINUE
      IF s.success_rate < 0.30: CONTINUE  // disqualified (<30% success)

      q = s.quality_median ?? 0
      l = s.latency_median_ms ?? Infinity

      // Higher quality wins. On tie: lower latency wins.
      IF q > best.quality OR (q == best.quality AND l < best.latency):
        best = { quality: q, latency: l, competitor: c }

    winners[task_id] = best.competitor

  RETURN winners
```
